# Each test case includes:
        - case_id: A unique ID
        - type: One of unit_conversion, type_check, or required_field
        - data: The input to be tested
        (can be a primitive or a dict)
        - config: Optional configuration parameters (may be missing or incomplete)

# basic data
                    
        case_id:
            sorsz√°m
            string formatted int
            e.g.: "case_id": "001"
        type
            type of task
                string format
                types: unit conversion / type_check / required_field
                e.g.: "type": "unit_conversion"

        -- json format --


# unit_conversion
        data:
          value
            type: number/null
          unit
            type: string/null
            possible units: lb/g/kg
        
        config:
          expected_unit
            type: string
            possible expected_unit: lb/g/kg
          min:
            type: int

        
        
# type_check
        
        data:
            type: number, null, string
            possible type_checks : int/double/string/null
        config:
            expected_type:
                type: string
                possible expected_types: int, str, float



# required_field
        
        data:
            type: null, number, string (also empty)
        config:
          allow_null:
            type: bool
